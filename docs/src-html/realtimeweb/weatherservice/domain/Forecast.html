159            /**
160             * Internal constructor to create a Forecast from a json representation.
161             * @param map The raw json data that will be parsed.
162             * @return 
163             */
164        @SuppressWarnings("unchecked")
165            public Forecast(Map<String, Object> raw) {
166            // TODO: Check that the data has the correct schema.
167            // NOTE: It's much safer to check the Map for fields than to catch a runtime exception.
168            try {
169                this.longDescription = ((Map<String, Object>) raw.get("data")).get("text").toString();
170                this.description = ((Map<String, Object>) raw.get("data")).get("weather").toString();
171                this.imageUrl = ((Map<String, Object>) raw.get("data")).get("iconLink").toString();
172                this.temperatureLabel = ((Map<String, Object>) raw.get("time")).get("tempLabel").toString();
173                this.periodName = ((Map<String, Object>) raw.get("time")).get("startPeriodName").toString();
174                this.probabilityOfPrecipitation = Integer.parseInt(((Map<String, Object>) raw.get("data")).get("pop").toString());
175                this.periodTime = ((Map<String, Object>) raw.get("time")).get("startValidTime").toString();
176                this.temperature = Integer.parseInt(((Map<String, Object>) raw.get("data")).get("temperature").toString());
177            } catch (NullPointerException e) {
178                    System.err.println("Could not convert the response to a Forecast; a field was missing.");
179                    e.printStackTrace();
180            } catch (ClassCastException e) {
181                    System.err.println("Could not convert the response to a Forecast; a field had the wrong structure.");
182                    e.printStackTrace();
183            }
184        
185            }

198            @SuppressWarnings("unchecked")
199            public static ArrayList<Forecast> parse(Map<String, Object> raw) {
200                    HashMap<String, Object> rawTime = (HashMap<String, Object>) raw.get("time");
201                    HashMap<String, Object> rawData = (HashMap<String, Object>) raw.get("data");
202                    
203                    ArrayList<String> periodNames = ((ArrayList<String>)rawTime.get("startPeriodName"));
204                    ArrayList<String> periodTimes = ((ArrayList<String>)rawTime.get("startValidTime"));
205                    ArrayList<String> tempLabels = ((ArrayList<String>)rawTime.get("tempLabel"));
206                    ArrayList<String> temperatures = ((ArrayList<String>)rawData.get("temperature"));
207                    ArrayList<String> pops = ((ArrayList<String>)rawData.get("pop"));
208                    for (int i = 0; i < pops.size(); i+= 1) {
209                            if (pops.get(i) == null) {
210                                    pops.set(i, "0");
211                            }
212                    }
213                    ArrayList<String> weather = ((ArrayList<String>)rawData.get("weather"));
214                    ArrayList<String> icons = ((ArrayList<String>)rawData.get("iconLink"));
215                    ArrayList<String> text = ((ArrayList<String>)rawData.get("text"));
216    
217                    ArrayList<Forecast> forecasts = new ArrayList<Forecast>();
218                    for (int i = 0; i < periodNames.size(); i+= 1) {
219                            forecasts.add(new Forecast(periodNames.get(i), periodTimes.get(i), tempLabels.get(i), 
220                                            Integer.parseInt(temperatures.get(i)), Integer.parseInt(pops.get(i)), weather.get(i), icons.get(i), text.get(i)));
221                    }
222                    return forecasts;
223            }
