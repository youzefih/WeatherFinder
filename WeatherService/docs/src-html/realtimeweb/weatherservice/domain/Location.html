013    /**
014     * A detailed description of a location
015     */
016    public class Location {
017            
018        private Double latitude;
019        private Integer elevation;
020        private String name;
021        private Double longitude;
022        
023        
024        /**
025         * @return The latitude (up-down) of this location.
026         */
027        public Double getLatitude() {
028            return this.latitude;
029        }
030        
031        /**
032         * @param The latitude (up-down) of this location.
033         * @return Double
034         */
035        public void setLatitude(Double latitude) {
036            this.latitude = latitude;
037        }
038        
039        /**
040         * @return The height above sea-level (in feet).
041         */
042        public Integer getElevation() {
043            return this.elevation;
044        }
045        
046        /**
047         * @param The height above sea-level (in feet).
048         * @return Integer
049         */
050        public void setElevation(Integer elevation) {
051            this.elevation = elevation;
052        }
053        
054        /**
055         * @return The city and state that this location is in.
056         */
057        public String getName() {
058            return this.name;
059        }
060        
061        /**
062         * @param The city and state that this location is in.
063         * @return String
064         */
065        public void setName(String name) {
066            this.name = name;
067        }
068        
069        /**
070         * @return The longitude (left-right) of this location.
071         */
072        public Double getLongitude() {
073            return this.longitude;
074        }
075        
076        /**
077         * @param The longitude (left-right) of this location.
078         * @return Double
079         */
080        public void setLongitude(Double longitude) {
081            this.longitude = longitude;
082        }
083        
084            
085            /**
086             * Creates a string based representation of this Location.
087            
088             * @return String
089             */
090            public String toString() {
091                    return "Location[" +latitude+", "+elevation+", "+name+", "+longitude+"]";
092            }
093            
094            /**
095             * Internal constructor to create a Location from a json representation.
096             * @param map The raw json data that will be parsed.
097             * @return 
098             */
099        public Location(Map<String, Object> raw) {
100            // TODO: Check that the data has the correct schema.
101            // NOTE: It's much safer to check the Map for fields than to catch a runtime exception.
102            try {
103                this.latitude = Double.parseDouble(raw.get("latitude").toString());
104                this.elevation = Integer.parseInt(raw.get("elevation").toString());
105                this.name = raw.get("areaDescription").toString();
106                this.longitude = Double.parseDouble(raw.get("longitude").toString());
107            } catch (NullPointerException e) {
108                    System.err.println("Could not convert the response to a Location; a field was missing.");
109                    e.printStackTrace();
110            } catch (ClassCastException e) {
111                    System.err.println("Could not convert the response to a Location; a field had the wrong structure.");
112                    e.printStackTrace();
113            }
114        
115            }
116        
117        public static Location fromGeocode(Map<String, Object> raw) {
118            try {
119                    Map<String, Object> firstResult = (Map<String, Object>) ((List<Object>)raw.get("results")).get(0);
120                    Map<String, Object> location = (Map<String, Object>) ((Map<String, Object>)firstResult.get("geometry")).get("location");
121                Double latitude = Double.parseDouble(location.get("lat").toString());
122                Double longitude = Double.parseDouble(location.get("lng").toString());
123                String name = (String) firstResult.get("formatted_address");
124                return new Location(latitude, longitude, name, 0);
125            } catch (NullPointerException e) {
126                    System.err.println("Could not convert the geocode response to a Location; a field was missing.");
127                    e.printStackTrace();
128            } catch (ClassCastException e) {
129                    System.err.println("Could not convert the geocode response to a Location; a field had the wrong structure.");
130                    e.printStackTrace();
131            }
132                    return null;
133        }
134        
135        public Location(double latitude, double longitude, String name, int elevation) {
136            this.latitude = latitude;
137            this.longitude = longitude;
138            this.name = name;
139            this.elevation = elevation;
140        }
141    }
